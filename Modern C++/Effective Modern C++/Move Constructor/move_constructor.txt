#include <iostream>
#include <vector>
using namespace std;

struct test{
    string s;
    vector<int> vec;
    int* a;
    test(){
        cout<<"Default constructor called\n";
        s="Hello";
        a=new int[10];
        for(int i=0;i<10;i++)
            a[i]=i;
        vec={1,2,3,4};
    }
    test(const test& t){
        cout<<"Copy constructor called\n";
        s=t.s;
        vec=t.vec;
        a=new int[10];
        for(int i=0;i<10;i++){
            a[i]=t.a[i];
        }
    }
    test(test&& t):s(move(t.s)){
        cout<<"Move constructor called\n";
        vec=std::move(t.vec);
        a=t.a;
        t.a=nullptr;
    }
    
};

test func(test t){
    return t;
}
int main(){
    cout<<"Hello World\n";
    test t1;
    std::cout<<t1.vec[2];
    test t2=func(move(t1));
    cout<<"t2.s"<<t2.s<<t2.vec[3]<<endl;
    cout<<t2.a[5];
    return 0;
}
